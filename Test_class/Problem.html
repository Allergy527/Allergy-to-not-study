  • 那么正确的代码应该是在scanf()后面添加一个getchar()，来读取换行符避免gets()直接读入\0，进而导致程序后面的出错：
#include<stdio.h>

typedef struct stuInfo {
    char Num[10];
    char Name[10];
    float Eng, Math, Chin;
}stu;

int main() {
    stu qwq[5];
    float sum[5];
    for(int i = 0;i < 5;++i) {
        gets(qwq[i].Num);
        gets(qwq[i].Name);
        scanf("%f%f%f", &qwq[i].Eng, &qwq[i].Math, &qwq[i].Chin);
        char s = getchar(); //我们需要加上这个来帮助scanf读掉缓存区的的\n
        sum[i] = qwq[i].Eng + qwq[i].Math + qwq[i].Chin;
    }
    for(int i = 0;i < 5;++i)printf("%f\t", sum[i]);
    getchar();getchar();
    return 0;
}
